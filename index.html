href: string;
children: React.ReactNode;
variant?: "primary" | "secondary";

"inline-flex items-center justify-center rounded-2xl px-5 py-3 text-
sm font-semibold transition will-change-transform";

variant === "primary"

<a className={`${base} ${styles}`} href={href}>
{children}
</a>
);
